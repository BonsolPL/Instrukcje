#include <iostream>
using namespace std;

int main() {
    double suma = 0.0, liczba;

    while (true) {
        cout << "Podaj liczbe do sumowania: ";
        cin >> liczba;

        if (liczba == 0) {
            cout << "Nastapil BREAK" << endl;
            break;
        }

        if (liczba < 0) {
            continue;
        }

        suma += liczba;
        cout << "SUMA CZESCIOWA: " << suma << endl;
    }

    return 0;
}
