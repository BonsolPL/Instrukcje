#include <iostream>
using namespace std;

int main() {
    int miesiac;
    
    cout << "Podaj miesiac Twojego urodzenia: ";
    cin >> miesiac;
    
    int start = miesiac;
    if (miesiac % 2 != 0) {
        start = miesiac + 1;
    }
    
    for (int i = start; i <= miesiac + 30; i += 2) {
        cout << i << " ";
    }
    
    return 0;
}
